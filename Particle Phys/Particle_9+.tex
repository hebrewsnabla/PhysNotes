\begin{equation}\label{key}

\begin{equation}\label{key}

\end{equation}

\begin{equation}\label{key}

\end{equation}

\begin{equation}\label{key}

\begin{equation}\label{key}
[\hH, \hG] = 0
\end{equation}
thus
\begin{equation}\label{key}
\dv{t}\ev{\hG} = \I\ev{[\hH, \hG]} = 0 \;(\text{conserved})
\end{equation}
